function bookRide(){ const pickup=document.getElementById('pickup').value.trim(); const dest=document.getElementById('dest').value.trim(); const type=document.getElementById('ride-type').value; const uid=parseInt(document.getElementById('uid-book').value); if(!pickup || !dest){ notify('⚠ Enter pickup & destination',true); return; } const user=users.find(u=>u.id===uid); if(!user){ notify('⚠ User ID not found',true); return; } let avail = drivers.filter(d => d.available && (type==='Any' || d.vehicleType===type) ); if(avail.length===0){ notify('⚠ No drivers available',true); return; } const driverBox = document.getElementById('driver-options'); const driverList = document.getElementById('driver-list'); driverBox.style.display = 'block'; driverList.innerHTML = ''; avail.forEach(driver => { driverList.innerHTML += `
${driver.name}
${driver.vehicle} (${driver.vehicleType})
⭐ ${driver.rating} | ${driver.distanceKm} km away
`; }); }
